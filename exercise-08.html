Step 3 - derive by hand the oridinary least squares regression coefficients beta1 and beta 0 for ECV as a function of social group size


olsd <- d %>%
  drop_na(ECV) %>%
  drop_na(Group_size)

beta1 <- sum((olsd$ECV - mean(olsd$ECV))* (olsd$Group_size - mean(olsd$Group_size)))/sum((olsd$Group_size - mean(olsd$Group_size))^2)

beta1  #[1] 2.463071

beta0 <- mean(olsd$ECV) - beta1 * mean(olsd$Group_size)

Step 7 Use a permutation approach with 1000 permutations to generate a null sampling distribution for the slope coefficient. What is it that you need to permute?


nperm <- 1000
perm_coefs <- vector(length = nperm)

for (i in 1:nperm){
  d_perm <- olsd #make a fresh copy each iteration 
  d_perm$Group_size <- sample(olsd$Group_size) #shuffle predictor
  perm_coefs[[i]] <- lm(ECV ~ Group_size, data = d_perm) %>%
    tidy() %>%
    filter(term == "Group_size") %>%
    pull(estimate)
}

perm_coefs

observed_coef <- tidy(m)%>%
  filter(term == "Group_size")%>%
  pull(estimate)


p_value <- mean(abs(perm_coefs) >= abs(observed_coef))

cat("Permutation p-value:", p_value, "\n")

mean(abs(perm_coefs))
